Module shinyant.pagination

Pagination component module for shinyant UI library.

This module provides an Ant Design-inspired pagination component with flexible configuration options.

Key features include:

  • Responsive layout with adjustable alignment
  • Customizable page size options
  • Compact/simple modes for space efficiency
  • Quick jumper for direct page navigation
  • Size changer for dynamic items per page
  • Design tokens for full visual customization
  • etc.

The component follows Ant Design specifications while adapting them for Python usage.

Example

from shiny import App, reactive, render, ui

import shinyant


app_ui = ui.page_auto(
    shinyant.nesting_wrap(
        shinyant.pagination(
            id="pagination", total=500, current=3, showQuickJumper=True, showSizeChanger=True, responsive=True
        )
    ),
    ui.p("Current page: ", ui.output_text("current_page")),
)


def server(input, output, session):
    page_current = reactive.Value(3)

    @reactive.effect
    @reactive.event(input.pagination)
    async def update_pagination_info():
        if input.pagination():
            if "page" in input.pagination():
                page_current.set(input.pagination()["page"])

    @output
    @render.text
    def current_page():
        return f"{page_current.get()}"


app = App(app_ui, server)

def pagination(
    id: str,
    total: int = 0,
    align: PaginationAlignTypes = "",
    current: int | None = 0,
    defaultCurrent: int = 1,
    defaultPageSize: int = 10,
    disabled: bool = False,
    hideOnSinglePage: bool = False,
    pageSize: int | None = 10,
    pageSizeOptions: list[int] = [10, 20, 50, 100],
    responsive: bool = False,
    showLessItems: bool = False,
    showQuickJumper: bool | dict = False,
    showSizeChanger: bool | dict = False,
    showTitle: bool = True,
    simple: bool | dict = False,
    size: PaginationSizeTypes = "default",
    design: PaginationDesignTokens | None = None,
    globalDesign: PaginationGlobalTokens | None = None,
) -> "Pagination":
    """
    Creates a pagination component based on ant design react Pagination component

    Args:
        id (str): name of the component in the application
        total (int): Total number of data items. Defaults to 0
        align (str): Specify the position of the pagination. Defaults to ''
        current (int | None): Current page number. Defaults to None
        defaultCurrent (int): Default current page. Defaults to 1
        defaultPageSize (int): Default number of data items per page. Defaults to 10
        disabled (bool): Disabled state of pagination. Defaults to False
        hideOnSinglePage (bool): Whether to hide pagination on single page. Defaults to False
        pageSize (int | None): Number of data items per page. Defaults to None
        pageSizeOptions (list[int]): Specify the sizeChanger options. Defaults to [10, 20, 50, 100]
        responsive (bool): If pagination should be responsive. Defaults to False
        showLessItems (bool): Show less page items. Defaults to False
        showQuickJumper (bool | dict): Determine whether you can jump to pages directly. Defaults to False
        showSizeChanger (bool | dict): Determine whether to show pageSize select. Defaults to False
        showTitle (bool): Show page item's title. Defaults to True
        simple (bool | dict): Whether to use simple mode. Defaults to False
        size (str): Specify pagination size. Defaults to 'default'
        design (PaginationDesignTokens | None): Design of pagination. Defaults to None
        globalDesign (PaginationGlobalTokens | None): Global design of pagination. Defaults to None
    """
    if design is None:
        design = PaginationDesignTokens()
    if globalDesign is None:
        globalDesign = PaginationGlobalTokens()

    return Pagination(
        id=id,
        align=align,
        current=current,
        defaultCurrent=defaultCurrent,
        defaultPageSize=defaultPageSize,
        disabled=disabled,
        hideOnSinglePage=hideOnSinglePage,
        pageSize=pageSize,
        pageSizeOptions=pageSizeOptions,
        responsive=responsive,
        showLessItems=showLessItems,
        showQuickJumper=showQuickJumper,
        showSizeChanger=showSizeChanger,
        showTitle=showTitle,
        simple=simple,
        size=size,
        total=total,
        design=design,
        globalDesign=globalDesign,
    )
