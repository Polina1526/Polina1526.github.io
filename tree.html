def tree(
    id: str,
    autoExpandParent: bool = False,
    blockNode: bool = False,
    checkable: bool = False,
    checkStrictly: bool = False,
    defaultCheckedKeys: list[str] | None = None,
    defaultExpandAll: bool = False,
    defaultExpandedKeys: list[str] | None = None,
    defaultExpandParent: bool = True,
    defaultSelectedKeys: list[str] | None = None,
    disabled: bool = False,
    draggable: bool = False,
    multiple: bool = False,
    selectable: bool = True,
    showIcon: bool = False,
    showLine: bool = False,
    treeData: list[Any] | None = None,
    virtual: bool = True,
    design: TreeDesignTokens | None = None,
    globalDesign: TreeGlobalTokens | None = None,
):
    """
    Creates a tree component based on ant design react Tree component

    Attributes:
        id (str): name of the component in the application
        autoExpandParent (bool): Whether to automatically expand a parent treeNode. Defaults to False
        blockNode (bool): Whether treeNode fill remaining horizontal space. Defaults to False
        checkable (bool): Add a Checkbox before the treeNodes. Defaults to False
        checkedKeys (list[str]): Specifies the keys of the checked treeNodes. Defaults to []
        checkStrictly (bool): Check treeNode precisely; parent treeNode and children treeNodes are not associated. Defaults to False
        defaultCheckedKeys (list[str]): Specifies the keys of the default checked treeNodes. Defaults to []
        defaultExpandAll (bool): Whether to expand all treeNodes by default. Defaults to False
        defaultExpandedKeys (list[str]): Specify the keys of the default expanded treeNodes. Defaults to []
        defaultExpandParent (bool): If auto expand parent treeNodes when init. Defaults to True
        defaultSelectedKeys (list[str]): Specifies the keys of the default selected treeNodes. Defaults to []
        disabled (bool): Whether disabled the tree. Defaults to False

        draggable (bool): Specifies whether this Tree or the node is draggable. Defaults to False \n
         doesn't compatible with change_treeData function, change_treeData with draggable=True will have no effect,
         because we can't sync shiny with react when changes happen there and not in shiny. Also, shiny side will store initial
         treeData and you can't get actual treeData right now, but still can get selected and checked keys using input in server.
         TODO: implement sending the same JSON structure back to shiny

        expandedKeys (list[str]): (Controlled) Specifies the keys of the expanded treeNodes. Defaults to all keys
        multiple (bool): Allows selecting multiple treeNodes. Defaults to False
        selectable (bool): Whether can be selected. Defaults to True
        selectedKeys (list[str]): (Controlled) Specifies the keys of the selected treeNodes, multiple selection needs to set multiple to true. Defaults to all keys
        showIcon (bool): Controls whether to display the icon node, no default style. Defaults to False
        showLine (bool): Shows a connecting line. Defaults to False
        treeData (list[Any]): The treeNodes data Array. Defaults to []
        virtual (bool): Disable virtual scroll when set to false. Defaults to True
        design (TreeDesignTokens | None): Design of tree (TreeDesignTokens class object). Defaults to None
        globalDesign (TreeGlobalTokens | None): Global Design of tree (TreeGlobalTokens class object). Defaults to None
    """

    if design is None:
        design = TreeDesignTokens()
    if globalDesign is None:
        globalDesign = TreeGlobalTokens()
    if defaultCheckedKeys is None:
        defaultCheckedKeys = []
    if defaultExpandedKeys is None:
        defaultExpandedKeys = []
    if defaultSelectedKeys is None:
        defaultSelectedKeys = []
    if treeData is None:
        treeData = []

    return Tree(
        id,
        autoExpandParent,
        blockNode,
        checkable,
        checkStrictly,
        defaultCheckedKeys,
        defaultExpandAll,
        defaultExpandedKeys,
        defaultExpandParent,
        defaultSelectedKeys,
        disabled,
        draggable,
        multiple,
        selectable,
        showIcon,
        showLine,
        treeData,
        virtual,
        design,
        globalDesign,
    )

    """

    __trees: ClassVar[dict[str, "Tree"]] = {}

        self.defaultExpandAll = defaultExpandAll

        self.multiple = multiple

        self.showIcon = showIcon
        self.showLine = showLine

        self.virtual = virtual
        self.design = design
        self.globalDesign = globalDesign
        if resolve_id(id) in Tree.__trees:
            raise KeyError("'Tree' object with this id already exists")
        Tree.__trees[resolve_id(id)] = self

class TreeNode:
    """
    TreeNode Class (one node of Tree)

    Attributes:
        key (str): Must be unique in all of treeNodes of the tree
        title (str): Title
        checkable (bool): When Tree is checkable, set TreeNode display Checkbox or not. Defaults to False
        disableCheckbox (bool): Disables the checkbox of the treeNode. Defaults to False
        disabled (bool): Disables the treeNode. Defaults to False
        icon: (str): Customize icon. Defaults to ""
        selectable (bool): Set whether the treeNode can be selected. Defaults to True
    """

    key: str
    title: Any
    children: list["TreeNode"] = field(default_factory=list)
    checkable: bool = False
    disableCheckbox: bool = False
    disabled: bool = False
    selectable: bool = True

    def to_dict(self) -> dict[str, Any]:
        """
        Recursively builds a dictionary
        """

        return {
            "key": self.key,
            "title": self.title,
            "checkable": self.checkable,
            "disableCheckbox": self.disableCheckbox,
            "disabled": self.disabled,
            "selectable": self.selectable,
            "children": [child.to_dict() for child in self.children],
        }
