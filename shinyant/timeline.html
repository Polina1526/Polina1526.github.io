def timeline(
    id: str,
    mode: str = "left",
    pending: bool = False,
    reverse: bool = False,
    items: list[TimelineItem] | None = None,
    design: TimelineDesignTokens | None = None,
    globalDesign: TimelineGlobalTokens | None = None,
) -> "Timeline":
    """
    Creates a timeline component based on ant design react Timeline component

    Attributes:
        id (str): name of the component in the application
        mode (str): distibution of nodes (left | right | alternate). Defaults to 'left'
        pending (bool): Set the last ghost node's existence. Defaults to False
        reverse (bool): Whether reverse nodes or not. Defautls to False
        items (list[TimelineItem]): List of nodes of timeline (TimelineItem class objects). Defaults to None
        design (TimelineDesignTokens): Design of timeline (TimelineDesignTokens class object). Defaults to None
        globalDesign (TimelineGlobalTokens): Global design of timeline (TimelineGlobalTokens class object). Defaults to None
    """
    if items is None:
        items = []
    if design is None:
        design = TimelineDesignTokens()
    if globalDesign is None:
        globalDesign = TimelineGlobalTokens()

    return Timeline(id, mode, pending, reverse, items, design, globalDesign)

class Timeline:
    """
    Main class, stores all timelines and provides an interface to access them
    """

    __timelines: ClassVar[dict[str, "Timeline"]] = {}

    def __init__(
        self,
        id: str,
        mode: str,
        pending: bool,
        reverse: bool,
        items: list[TimelineItem],
        design: TimelineDesignTokens,
        globalDesign: TimelineGlobalTokens,
    ):
        self.id = resolve_id(id)
        self.mode = mode
        self.pending = pending
        self.reverse = reverse
        self.items = items
        self.design = design
        self.globalDesign = globalDesign
        if resolve_id(id) in Timeline.__timelines:
            raise KeyError("'Timeline' object with this id already exists")
        Timeline.__timelines[resolve_id(id)] = self

    @classmethod
    def get_component(cls, id: str) -> "Timeline":
        return Timeline.__timelines[resolve_id(id)]

    @classmethod
    async def change_mode(cls, id: str, mode: str) -> None:
        Timeline.__timelines[resolve_id(id)].mode = mode
        await send_data(resolve_id(id))

    @classmethod
    async def change_pending(cls, id: str, pending: bool) -> None:
        Timeline.__timelines[resolve_id(id)].pending = pending
        await send_data(resolve_id(id))

    @classmethod
    async def change_reverse(cls, id: str, reverse: bool) -> None:
        Timeline.__timelines[resolve_id(id)].reverse = reverse
        await send_data(resolve_id(id))

    @classmethod
    async def change_items(cls, id: str, items: list[TimelineItem]) -> None:
        Timeline.__timelines[resolve_id(id)].items = items
        await send_data(resolve_id(id))

    @classmethod
    async def change_design(cls, id: str, design: TimelineDesignTokens) -> None:
        Timeline.__timelines[resolve_id(id)].design = design
        await send_data(resolve_id(id))

    @classmethod
    async def change_global_design(cls, id: str, globalDesign: TimelineGlobalTokens) -> None:
        Timeline.__timelines[resolve_id(id)].globalDesign = globalDesign
        await send_data(resolve_id(id))

    def dump(self) -> dict:
        data = {
            "shinyant_timeline": {
                "id": self.id,
                "mode": self.mode,
                "pending": self.pending,
                "reverse": self.reverse,
                "items": [{key: value for key, value in asdict(item).items() if value != ""} for item in self.items],
                "design": asdict(self.design),
                "globalDesign": asdict(self.globalDesign),
            }
        }
        return data
