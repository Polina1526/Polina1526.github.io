Module shinyant.input

Input component module for shinyant UI library.

This module provides an Ant Design-inspired input component with various types, styles and configurations.

Key features include:

  • Multiple input types (basic, password, textArea, search)

  • Customizable design tokens for styling

  • Support for addons, prefixes/suffixes

  • Validation states (error, warning)

  • Responsive sizing and variants

  • Character counting functionality

  • etc.

The component follows Ant Design specifications while adapting them for Python usage.

Example

from shiny import App, reactive, render, ui

import shinyant


app_ui = ui.page_auto(
    ui.p("Input example"),
    shinyant.nesting_wrap(shinyant.input(id="input", placeholder="Input text", allowClear=True)),
    ui.p("Output shiny:"),
    ui.output_text(id="input_text"),
)


def server(input, output, session):
    @render.text
    @reactive.event(input.input)
    def input_text():
        return input.input()


app = App(app_ui, server)

def input(
    id: str,
    type: InputTypes = "basic",
    defaultValue: str = "",
    placeholder: str | None = "",
    addonAfter: str | list[str] = "",
    addonBefore: str | list[str] = "",
    allowClear: bool = False,
    autoSize: bool | dict[str, int] | None = None,
    count: InputCountConfig | None = None,
    disabled: bool = False,
    enterButton: bool | None = False,
    loading: bool | None = False,
    maxLength: int | None = None,
    prefix: str = "",
    showCount: bool = False,
    status: StatuaTypes | None = None,
    size: SizeTypes = "middle",
    suffix: str = "",
    variant: VariantTypes = "outlined",
    design: InputDesignTokens | None = None,
    globalDesign: InputGlobalTokens | None = None,
) -> "Input":
    """
    Creates a input component based on ant design react Input component

    Attributes:
        id (str): name of the component in the application
        type (str): Input types: 'basic', 'password', 'textArea'. Defauls to 'basic'
        defaultValue (str): The initial input content. Defauls to ''
        placeholder (str): Placeholder to display in empty input field. Defauls to ''
        addonAfter (str | list[str]): The label text displayed after (on the right side of) the input field.
            Use a list[str] to manage addonAfter as SelectAfter. Defauls to ''
        addonBefore (str | list[str]): The label text displayed before (on the left side of) the input field.
            Use a list[str] to manage addonBefore as SelectBefore. Defauls to ''
        allowClear (bool): If allow to remove input content with clear icon. Defauls to False
        autoSize (bool | dict[str, int]): Whether to resize automatically (True/False), you can specify autoSize={"minRows": 2, "maxRows": 4} to specify resize limits.
            Works only with parameter type='textArea'. Defauls to None
        count (InputCountConfig): Character count config, for more info see documentation for InputCountConfig. Defauls to None
        disabled (bool): Whether the input is disabled. Defauls to False
        enterButton (bool): False - displays the default button color, True - uses the primary color. Conflicts with addonAfter. Defauls to False
        loading (bool): Search box with loading. Defauls to False
        maxLength (int): The maximum number of characters in Input. Defauls to None
        prefix (str): The prefix string for the Input. Defauls to ''
        showCount (bool): Whether to show character count. Defauls to False
        status (str | None): Set validation status  "error" | "warning". Defauls to None
        size (str): The size of the input box. Note: in the context of a form, the middle size is used. Defauls to 'middle'
        suffix (str): The suffix string for the Input. Defauls to ''
        variant (str): Variants of Input. Defauls to 'outlined'
        design (InputDesignTokens): Design of input (InputDesignTokens class object). Defauls to InputDesignTokens
        globalDesign (InputGlobalTokens): Global design of input (InputGlobalTokens class object). Defauls to InputGlobalTokens
    """
    if design is None:
        design = InputDesignTokens()
    if globalDesign is None:
        globalDesign = InputGlobalTokens()

    if type != "textArea" and autoSize is not None:
        autoSize = None

    if type != "search" and (enterButton is not None or loading is not None):
        enterButton = None
        loading = None

    value: str = defaultValue or ""

    return Input(
        id=id,
        type=type,
        addonAfter=addonAfter,
        addonBefore=addonBefore,
        allowClear=allowClear,
        autoSize=autoSize,
        count=count,
        disabled=disabled,
        maxLength=maxLength,
        enterButton=enterButton,
        loading=loading,
        placeholder=placeholder,
        prefix=prefix,
        showCount=showCount,
        status=status,
        size=size,
        suffix=suffix,
        value=value,
        variant=variant,
        design=design,
        globalDesign=globalDesign,
    )

@dataclass
class InputGlobalTokens:
    """
    Input Global Design class (Global Tokens from react ant.design)

    Attributes:
        colorBgContainerDisabled (str): Control the background color of container in disabled state
        colorBorder (str): Default border color, used to separate different elements, such as: form separator, card separator, etc
        colorError (str): Used to represent the visual elements of the operation failure, such as the error Button, error Result component, etc
        colorErrorBg (str): The background color of the error state
        colorErrorText (str): The default state of the text in the error color
        colorFillTertiary (str): The third level of fill color is used to outline the shape of the element, such as Slider, Segmented, etc. If there is no emphasis requirement, it is recommended to use the third level of fill color as the default fill color
        colorPrimaryActive (str): Dark active state under the main color gradient
        colorPrimaryHover (str): Hover state under the main color gradient
        colorSplit (str): Used as the color of separator, this color is the same as colorBorderSecondary but with transparency
