def calendar(
    id: str,
    defaultValue: str | None = None,
    fullscreen: bool = False,
    showWeek: bool = False,
    locale: LocaleType = "enUS",
    mode: ModeType = "month",
    validRange: list[str] | None = None,
    design: CalendarDesignTokens | None = None,
    globalDesign: CalendarGlobalTokens | None = None,
) -> "Calendar":
    """
    Creates a input component based on ant design react Input component

    Attributes:
        id (str): name of the component in the application
        defaultValue (str): The date selected by default. Defauls to ''
        fullscreen (bool): Whether to display in full-screen. Defauls to False
        showWeek (bool): Whether to display week number. Defauls to False
        locale (str): The calendar's locale: 'enUS', 'ruRU'. Defauls to "enUS"
        mode (str): The display mode of the calendar. Defauls to 'month'
        validRange (list[str]): To set valid range: list with start and end range date. Defauls to None
        design (CalendarDesignTokens): Design of input (CalendarDesignTokens class object). Defauls to CalendarDesignTokens
        globalDesign (CalendarGlobalTokens): Global design of input (CalendarGlobalTokens class object). Defauls to CalendarGlobalTokens
    """
    if design is None:
        design = CalendarDesignTokens()
    if globalDesign is None:
        globalDesign = CalendarGlobalTokens()

    value: str = defaultValue or ""

    return Calendar(
        id=id,
        fullscreen=fullscreen,
        showWeek=showWeek,
        locale=locale,
        mode=mode,
        validRange=validRange,
        value=value,
        design=design,
        globalDesign=globalDesign,
    )

class Calendar:
    """
    Main class, stores all calendar and provides an interface to access them
    """

    __calendars: ClassVar[dict[str, "Calendar"]] = {}

    def __init__(
        self,

        fullscreen: bool,
        showWeek: bool,
        locale: LocaleType,
        mode: ModeType,
        validRange: list[str] | None,
        value: str,
        design: CalendarDesignTokens,
        globalDesign: CalendarGlobalTokens,
    ):
        self.id = resolve_id(id)
        self.fullscreen = fullscreen
        self.showWeek = showWeek
        self.locale = locale
        self.mode = mode
        self.validRange = validRange
        self.value = value
        self.design = design
        self.globalDesign = globalDesign
        if resolve_id(id) in Calendar.__calendars:
            raise KeyError("'Calendar' object with this id already exists")
        Calendar.__calendars[resolve_id(id)] = self
