def breadcrumb(
    id: str,
    items: list[BreadcrumbItem] | None = None,
    separator: str = "/",
    design: BreadcrumbDesignTokens | None = None,
    globalDesign: BreadcrumbGlobalTokens | None = None,
) -> "Breadcrumb":
    """
    Creates a breadcrumb component based on ant design react Breadcrumb component

    Attributes:
        id (str): name of the component in the application
        items (list[BreadcrumbItem]): The routing stack information of router. Defauls to None
        separator (str): Custom separator. Set as '' to disable the common separator and enable customization of individual separators with BreadcrumbItem. Defauls to '/'
        design (BreadcrumbDesignTokens): Design of breadcrumb (BreadcrumbDesignTokens class object). Defauls to BreadcrumbDesignTokens
        globalDesign (BreadcrumbGlobalTokens): Global design of breadcrumb (BreadcrumbGlobalTokens class object). Defauls to BreadcrumbGlobalTokens
    """
    if items is None:
        items = []
    if design is None:
        design = BreadcrumbDesignTokens()
    if globalDesign is None:
        globalDesign = BreadcrumbGlobalTokens()

    return Breadcrumb(
        id=id,
        items=items,
        separator=separator,
        design=design,
        globalDesign=globalDesign,
    )

class Breadcrumb:
    """
    Main class, stores all breadcrumb and provides an interface to access them
    """

    __breadcrumb: ClassVar[dict[str, "Breadcrumb"]] = {}

    def __init__(
        self,
        id: str,
        items: list[BreadcrumbItem],
        separator: str,
        design: BreadcrumbDesignTokens,
        globalDesign: BreadcrumbGlobalTokens,
    ):
        self.id = resolve_id(id)
        self.items = items
        self.separator = separator
        self.design = design
        self.globalDesign = globalDesign
        if resolve_id(id) in Breadcrumb.__breadcrumb:
            raise KeyError("'Breadcrumb' object with this id already exists")
        Breadcrumb.__breadcrumb[resolve_id(id)] = self

    @classmethod
    def get_component(cls, id: str) -> "Breadcrumb":
        return Breadcrumb.__breadcrumb[resolve_id(id)]

    @classmethod
    async def change_items(cls, id: str, items: list[BreadcrumbItem]) -> None:
        Breadcrumb.__breadcrumb[resolve_id(id)].items = items
        await send_data(resolve_id(id))

    @classmethod
    async def change_separator(cls, id: str, separator: str) -> None:
        Breadcrumb.__breadcrumb[resolve_id(id)].separator = separator
        await send_data(resolve_id(id))

    @classmethod
    async def change_design(cls, id: str, design: BreadcrumbDesignTokens) -> None:
        Breadcrumb.__breadcrumb[resolve_id(id)].design = design
        await send_data(resolve_id(id))

    @classmethod
    async def change_global_design(cls, id: str, globalDesign: BreadcrumbGlobalTokens) -> None:
        Breadcrumb.__breadcrumb[resolve_id(id)].globalDesign = globalDesign
        await send_data(resolve_id(id))

    def dump(self) -> dict:
        data = {
            "shinyant_breadcrumb": {
                "id": self.id,
                "items": [{key: value for key, value in asdict(item).items() if value != ""} for item in self.items],
                "separator": self.separator,
                "design": asdict(self.design),
                "globalDesign": asdict(self.globalDesign),
            }
        }
        return data

        colorBgTextHover (str): Control the background color of text in hover state. Defaults to 'rgba(0,0,0,0.06)'
        colorPrimaryBorder (str): The stroke color under the main color gradient, used on the stroke of components such as Slider. Defaults to '#91caff'

Defaults to FontFamily
        fontSize (int): The most widely used font size in the design system, from which the text gradient will be derived. Defaults to 14
        fontSizeIcon (int): Control the font size of operation icon in Select, Cascader, etc. Normally same as fontSizeSM. Defaults to 12
        lineHeight (float): Line height of text. Defaults to 1.5714285714285714
        lineWidthFocus (int): Control the width of the line when the component is in focus state. Defaults to 3
        marginXXS (int): Control the margin of an element, with the smallest size. Defaults to 4
        motionDurationMid (str): Motion speed, medium speed. Used for medium element animation interaction. Defaults to '0.2s'
        paddingXXS (int): Control the extra extra small padding of the element. Defaults to 4
    """
