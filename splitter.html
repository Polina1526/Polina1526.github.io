Module shinyant.splitter

Splitter component module for shinyant UI library.

This module provides an Ant Design-inspired resizable split panel component with flexible layout options.

Key features include:

  • Horizontal and vertical layout modes

  • Configurable panels with adjustable sizes and constraints

  • Support for collapsible panels

  • Customizable design tokens for styling

  • Lazy loading of panel content

  • Drag-and-drop resizing functionality

  • etc.

The component follows Ant Design specifications while adapting them for Python usage.

Example

from shiny import App, ui

import shinyant


app_ui = ui.page_auto(
    shinyant.nesting_wrap(
        shinyant.splitter(
            id="main_splitter",
            layout="horizontal",
            panels=[
                shinyant.splitter_panel(
                    defaultSize="35%",
                    children=[
                        shinyant.splitter(
                            id="vertical_splitter",
                            layout="vertical",
                            panels=[
                                shinyant.splitter_panel(children=[ui.div("Top panel")]),
                                shinyant.splitter_panel(children=[ui.div("Bottom panel")]),
                            ],
                        )
                    ],
                ),
                shinyant.splitter_panel(children=[ui.div("Side panel")]),
            ],
        )
    ),
)


def server(input, output, session):
    pass


app = App(app_ui, server)

def splitter(
    id: str,
    layout: SplitterLayoutTypes = "horizontal",
    lazy: bool = False,
    panels: list[SplitterPanel] | None = None,
    design: SplitterDesignTokens | None = None,
    globalDesign: SplitterGlobalTokens | None = None,
) -> "Splitter":
    """
    Creates a splitter component based on ant design react Splitter component

    Args:
        id (str): name of the component in the application
        layout (str): Layout direction, can be 'horizontal' or 'vertical'. Defaults to 'horizontal'
        lazy (bool): Lazy mode for loading panel content. Defaults to False
        panels (list[SplitterPanel] | None): List of SplitterPanel objects. Defaults to None
        design (SplitterDesignTokens | None): Design tokens for customizing component appearance. Defaults to None
        globalDesign (SplitterGlobalTokens | None): Global design tokens for theme customization. Defaults to None
    """
    if design is None:
        design = SplitterDesignTokens()
    if globalDesign is None:
        globalDesign = SplitterGlobalTokens()
    if panels is None:
        panels = []

    return Splitter(
        id=id,
        layout=layout,
        lazy=lazy,
        panels=panels,
        design=design,
        globalDesign=globalDesign,
    )

def splitter_panel(
    defaultSize: int | str | None = None,
    min: int | None = None,
    max: int | None = None,
    size: int | None = None,
    collapsible: bool | dict = False,
    resizable: bool = True,
    children: list[Tag | object] | None = None,
) -> SplitterPanel:
    """
    Create a configuration for a Splitter.Panel to be used with the splitter component.

    Args:
        defaultSize (int | str | None): Initial panel size support number for px or 'percent%' usage. Defaults to None
        min (int | None): Minimum threshold in pixels. Defaults to None
        max (int | None): Maximum threshold in pixels. Defaults to None
        size (int | None): Controlled panel size in pixels. Defaults to None
        collapsible (bool | dict): Quick folding. Defaults to False
        resizable (bool): Whether to enable drag and drop. Defaults to True
        children (list): Child elements to render in the panel. Defaults to None

    Returns:
        A SplitterPanel object.
    """
    return SplitterPanel(
        defaultSize=defaultSize,
        min=min,
        max=max,
        size=size,
        collapsible=collapsible,
        resizable=resizable,
        children=children or [],
    )

class Splitter:
    """
    Main class, stores all splitters and provides an interface to access them
    """

    __splitters: ClassVar[dict[str, "Splitter"]] = {}

    def __init__(
        self,
        id: str,
        layout: SplitterLayoutTypes,
        lazy: bool = False,
        panels: list[SplitterPanel] | None = None,
        design: SplitterDesignTokens | None = None,
        globalDesign: SplitterGlobalTokens | None = None,
    ):
        self.id = resolve_id(id)
        self.layout = layout
        self.lazy = lazy
        self.panels = panels or []
        self.design = design or SplitterDesignTokens()
        self.globalDesign = globalDesign or SplitterGlobalTokens()

        if resolve_id(id) in Splitter.__splitters:
            raise KeyError("'Splitter' object with this id already exists")
        Splitter.__splitters[resolve_id(id)] = self

    @classmethod
    def get_component(cls, id: str) -> "Splitter":
        return Splitter.__splitters[resolve_id(id)]

    @classmethod
    async def change_layout(cls, id: str, layout: SplitterLayoutTypes) -> None:
        Splitter.__splitters[resolve_id(id)].layout = layout
        await send_data(resolve_id(id))

    @classmethod
    async def change_lazy(cls, id: str, lazy: bool) -> None:
        Splitter.__splitters[resolve_id(id)].lazy = lazy
        await send_data(resolve_id(id))

    @classmethod
    async def change_panels(cls, id: str, panels: list[SplitterPanel]) -> None:
        Splitter.__splitters[resolve_id(id)].panels = panels
        await send_data(resolve_id(id))

    @classmethod
    async def change_design(cls, id: str, design: SplitterDesignTokens) -> None:
        Splitter.__splitters[resolve_id(id)].design = design
        await send_data(resolve_id(id))

    @classmethod
    async def change_global_design(cls, id: str, globalDesign: SplitterGlobalTokens) -> None:
        Splitter.__splitters[resolve_id(id)].globalDesign = globalDesign
        await send_data(resolve_id(id))

    def dump(self) -> dict:
        """Dump splitter data for serialization"""
        data = {
            "shinyant_splitter": {
                "id": self.id,
                "layout": self.layout,
                "lazy": self.lazy,
                "panels": [panel.to_dict() for panel in self.panels],
                "design": asdict(self.design),
                "globalDesign": asdict(self.globalDesign),
            }
        }
        return data

class SplitterPanel:
    """
    Splitter Panel class, represents a panel in the Splitter component

    Attributes:
        defaultSize (int | str | None): Initial panel size support number for px or 'percent%' usage
        min (int | None): Minimum threshold in pixels
        max (int | None): Maximum threshold in pixels
        size (int | None): Controlled panel size in pixels
        collapsible (bool | dict): Quick folding
        resizable (bool): Whether to enable drag and drop
        children (list): Child elements to render in the panel
    """

    def __init__(
        self,
        defaultSize: int | str | None = None,
        min: int | None = None,
        max: int | None = None,
        size: int | None = None,
        collapsible: bool | dict = False,
        resizable: bool = True,
        children: list[Tag | object] | None = None,
    ):
        self.defaultSize = defaultSize
        self.min = min
        self.max = max
        self.size = size
        self.collapsible = collapsible
        self.resizable = resizable
        self.children = children or []

    def to_dict(self) -> dict[str, Any]:
        """
        Builds a dictionary representation of the panel
        """
        panel_data = {
            "defaultSize": self.defaultSize,
            "min": self.min,
            "max": self.max,
            "size": self.size,
            "collapsible": self.collapsible,
            "resizable": self.resizable,
        }

        if self.children:
            panel_data["children"] = self.children

        return panel_data
